plot_stage <- function(stage_name, data, ann_colors) {
  
  df <- data %>%
    filter(Stage == stage_name)
  
  ggplot(df, aes(x = Timepoint, y = log2CPM)) +
    
    geom_boxplot(
      aes(fill = Species, group = interaction(Timepoint, Species)),
      position = position_dodge(width = 0.8),
      alpha = 0.3,
      outlier.shape = NA
    ) +
    
    scale_fill_manual(values = ann_colors$species_cor) +
    
    geom_jitter(
      aes(color = Individual),
      position = position_jitterdodge(
        jitter.width = 0.15,
        dodge.width = 0.8
      ),
      size = 1,
      alpha = 0.6
    ) +
    
    scale_color_manual(values = ann_colors$individual_cor) +
    
    geom_line(
      stat = "summary",
      fun = mean,
      aes(group = Species, linetype = Species),
      position = position_dodge(width = 0.8),
      linewidth = 1
    ) +
    
    scale_linetype_manual(values = c(C = "dotted", H = "solid")) +
    
    geom_point(
      stat = "summary",
      fun = mean,
      aes(shape = Species),
      position = position_dodge(width = 0.8),
      size = 2
    ) +
    
    facet_wrap(~ Gene, scales = "free_y") +
    
    theme_classic() +
    theme(axis.text.x = element_text(angle = 45, hjust = 1)) +
    
    ggtitle(stage_name)
}

compute_frip_minimal <- function(bam_files, peak_files) {
  results <- lapply(seq_along(bam_files), function(i) {
    bam   <- bam_files[i]
    peaks <- peak_files[i]
    sample_name <- tools::file_path_sans_ext(basename(bam))

    reads <- readGAlignments(bam)

    pk <- import(peaks)
    pk <- reduce(pk)

    hits <- countOverlaps(reads, pk)
    reads_in_peaks <- sum(hits > 0)
    total_reads    <- length(reads)
    frip           <- reads_in_peaks / total_reads

    data.frame(
      sample         = sample_name,
      reads_in_peaks = reads_in_peaks,
      total_reads    = total_reads,
      FRiP           = frip
    )
  })

  bind_rows(results)
}

#BySpecies
qc_labeled <- qc_long %>%
  filter(metric %in% c("raw", "rep2", "rep3", "rep4")) %>%
  group_by(Species, metric) %>%
  mutate(
    q1 = quantile(value, 0.25, na.rm = TRUE),
    q3 = quantile(value, 0.75, na.rm = TRUE),
    iqr = q3 - q1,
    is_outlier = value < (q1 - 1.5 * iqr) |
                 value > (q3 + 1.5 * iqr)
  ) %>%
  ungroup()

qc_labeled %>%
  ggplot(aes(x = metric, y = value)) +
  geom_boxplot(outlier.shape = 3) +
  geom_jitter(width = 0.2, alpha = 0.6) +
  geom_text_repel(
    data = subset(qc_labeled, is_outlier),
    aes(label = SampleName),
    size = 3,
    max.overlaps = 50
  ) +
  facet_wrap(~Species) +
  theme_bw() +
  labs(
    x = "Peak set",
